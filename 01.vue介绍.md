### vue实例

![](./images/Jietu20171206-142233.jpg)

    <script src="https://vuejs.org/js/vue.js"></script>
    
    <div id="demo"> 
    	<span v-on:click = "clickHandel">{{message}}</span>
    </div>
    
    <script>
    	let obj = {
    		message : 'hello'
    	}
    
    	new Vue({
    		el : '#demo', //挂载的元素
    		data : obj,
    		methods : {
    			clickHandel () {
    				alert('span')
    			}
    		}
    	})
    </script>
